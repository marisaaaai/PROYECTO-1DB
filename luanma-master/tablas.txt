VREAR TABLAS

CREATE TABLE usuario_artista(usuarioArtista_id VARCHAR(30), correo VARCHAR(50),contrasena VARCHAR(70), nombre VARCHAR(50),
							PRIMARY KEY (usuarioArtista_id));
CREATE TABLE artista (artista_id SERIAL, nombre VARCHAR(50), PRIMARY KEY(artista_id));

CREATE TABLE album  (album_id SERIAL, nombre VARCHAR(50), artista_id SERIAL, PRIMARY KEY (album_id),
					FOREIGN KEY (artista_id) REFERENCES artista(artista_id) ON DELETE CASCADE ON UPDATE CASCADE);
--activada es 0 e inactivada 1
CREATE TABLE cancion (cancion_id SERIAL, nombre VARCHAR(50), enlace VARCHAR(50),genero VARCHAR(50), artista_id SERIAL,
					 album_id SERIAL, activada INT, PRIMARY KEY(cancion_id), FOREIGN KEY (artista_id) REFERENCES artista(artista_id) ON DELETE CASCADE ON UPDATE CASCADE,
					 FOREIGN KEY (album_id) REFERENCES album(album_id) ON DELETE CASCADE ON UPDATE CASCADE);
					 
CREATE TABLE administrador (administrador_id SERIAL, correo VARCHAR(50), contrasena VARCHAR(50), PRIMARY KEY (administrador_id));

CREATE TABLE usuario (usuario_id VARCHAR(50), correo VARCHAR(50), contrasena VARCHAR(50), nombreUsuario VARCHAR(50), suscripcion INT,
					 PRIMARY KEY (usuario_id));
CREATE TABLE reproducciones (cancion_id SERIAL, usuario_id VARCHAR(50), FOREIGN KEY (cancion_id) REFERENCES 
							cancion(cancion_id), FOREIGN KEY (usuario_id) REFERENCES usuario(usuario_id) ON DELETE CASCADE ON UPDATE CASCADE);

CREATE TABLE playlist (playlist_id SERIAL, nombre VARCHAR(50), usuario_id VARCHAR(50), primary key (playlist_id), 
					  FOREIGN KEY (usuario_id) REFERENCES usuario(usuario_id) ON DELETE CASCADE ON UPDATE CASCADE);
					  
CREATE TABLE canciones_playlist(playlist_id SERIAL, cancion_id SERIAL, FOREIGN KEY (playlist_id) REFERENCES
							   playlist(playlist_id) ON DELETE CASCADE ON UPDATE CASCADE, FOREIGN KEY (cancion_id) REFERENCES cancion(cancion_id) ON DELETE CASCADE ON UPDATE CASCADE);
